JS。

$.require("ready,./more/avalon", function() {
    var model = $.ViewModel({
        content: "<strong>钓鱼台</strong>是我们的"
    });
    $.View(model);
    var i = 1;
    setInterval(function() {
        if (i) {
            model.content("<strong>北方四岛</strong>也是我们的");
            i = 0;
        } else {
            model.content("<strong>钓鱼台</strong>是我们的");
            i = 1;
        }
    }, 1000);
});
