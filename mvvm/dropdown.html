dropdown

拥有下拉菜单的按钮。

参数：

parent
可选。String|Element。指定UI组件要添加的位置，没有默认为body
preRender
可选。Function。当插入DOM树前执行的函数，只执行一次
text
可选。String。按钮的文字
menu
可选。Array。可以是一个字符串数组，但框架会将它转为一个对象数组，将原字符串作为相应位置上的对象的text属性。如果本来就是对象数组， 对象要求有cls，text，href这三属性，没有框架会帮你补上。
split
可选。Boolean或0,1。即将三角形用一个独立的按钮放置
menucls
可选。String。决定下拉面板是相对按钮的左边对齐还是右边对齐，默认是左边。

除了通用成员外，此实例还拥有以下成员：

size
方法，我们可以传入"btn-mini"，"btn-small"， "btn-large"之中的任何一个作为参数，改变它的尺寸。

另外我们可以在instance.ui.addClass("dropup")，让菜单向上显示。

$.require("ready,./more/artifact/dropdown", function() {
    var opts = {
        btn_text: "下拉框",
        parent: ".btn-toolbar",
        menu: [{
            text: "子菜单1"
        }, {
            text: "子菜单2"
        }, {
            text: "子菜单3"
        }, {
            cls: "divider"
        }, {
            text: "被间隔的链接"
        }]
    }
    var btn = new $.ui.DropDown(opts);
    var blue_btn = new $.ui.DropDown(opts)
    blue_btn.VM.cls("btn-primary");
    var red_btn = new $.ui.DropDown(opts)
    red_btn.VM.cls("btn-danger")
    var yellow_btn = new $.ui.DropDown(opts)
    yellow_btn.VM.cls("btn-warning")
    var green_btn = new $.ui.DropDown(opts)
    green_btn.VM.cls("btn-success")
    var cyan_btn = new $.ui.DropDown(opts)
    cyan_btn.VM.cls("btn-info")
    var black_btn = new $.ui.DropDown(opts)
    black_btn.VM.cls("btn-inverse")
    new $.ui.DropDown({
        split: 1,
        menu: [1],
        parent: "#dropdown"
    })
    setTimeout(function() {
        blue_btn.VM.cls({
            "btn-large": 0
        })
    }, 500)
    setTimeout(function() {
        blue_btn.VM.cls("btn-large")
    }, 2000)
    setTimeout(function() {
        blue_btn.size("btn-small");
    }, 3000)
})
