$.String.unescapeHTML

描述：

将用escapeHTML方法处理过的字符串，还原为可以document解析为HTML标签的字符串。

        unescapeHTML: function (target) {
            return  target.replace(/"/g,'"')
            .replace(/</g,'<')
            .replace(/>/g,'>')
            .replace(/&/g, "&");
            //处理转义的中文和实体字符
            return target.replace(/&#([\d]+);/g, function(_0, _1){
                return String.fromCharCode(parseInt(_1, 10));
            });
        },
例子 
               $.require("lang",function($$){
                     var s = "<h1>>Pride & Prejudice</h1>>";
                    $.log($$(s).unescapeHTML(),true);
                    var s = "x > 10";
                    $.log($$(s).unescapeHTML(),true);
                });
