$.Array.intersect(arr,arr2)

描述：

求arr与arr2的交集。

参数：

arr2
必需。Array。
例子 
$.require("lang",function(){
    var a = [1, 2, 3, "a"];
    //var b = $.lang( a ).intersect([1, "a", 2]).value();
    var b = $.Array.intersect(a, [1, "a", 2]);//取交集
    var ret = $.dump( b );//序列化
    $.log( ret,true) //1,2,"a"
});
